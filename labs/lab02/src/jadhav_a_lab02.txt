    // Processing/Output

    if (alphabet == 'a' || alphabet == 'e' || alphabet == 'i' || alphabet == 'o' || alphabet == 'u')

            printf("%c is a vowel\n", alphabet);

    else

        {
            printf("%c is a vowel\n", alphabet);
        }
        else
        {
            printf("%c is not a vowel", alphabet);
        }

    return 0;

}
