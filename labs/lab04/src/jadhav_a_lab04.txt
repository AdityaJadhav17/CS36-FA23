void interchange_largest_smallest(int arr[], int size)
{

    int max_Index = 0;

    for (int i = 1; i < size; i++)
    {
        if (arr[i] > arr[max_Index])
        {
            max_Index = i;
        }
    }

    int min_Index = 0;

    for (int i = 1; i < size; i++)
    {
        if (arr[i] < arr[min_Index])
        {
            min_Index = i;
        }
    }

    int temp = arr[max_Index];
    arr[max_Index] = arr[min_Index];
    arr[min_Index] = temp;

}

{
    //Declaration

    int size;

    //Data Input

    printf("Enter the number of elements in the array: ");
    scanf("%d", &size);
    puts ("");

    int arr[size];

    //Output

    read_array(arr, size);

    printf("\nThe numbers you entered are: ");
    display_array(arr, size);

    int largest = arr[0];

    for (int i = 1; i < size; i++)
    {
        if (arr[i] > largest)
        {
            largest = arr[i];
        }
    }

    int secondLargest = find_second_largest(arr, size);

    printf("The largest of these numbers is: %d\n", largest);
    printf("The second largest of these numbers is: %d\n", secondLargest);

    return 0;
}


void read_array(int arr[], int size)
{
    for (int i = 0; i < size; i++)
    {
        printf("Enter the number: ");
        scanf("%d", &arr[i]);
    }
}


void display_array(int arr[], int size)
{
    for (int i = 0; i < size; i++)
    {
        printf("%d ", arr[i]);
    }
    puts("");
}


int find_second_largest(int arr[], int size)
{
    int largest = arr[0];
    int secondLargest = arr[0];

    for (int i = 1; i < size; i++)
    {
        if (arr[i] > largest)
        {
            secondLargest = largest;
            largest = arr[i];
        }
        else if (arr[i] > secondLargest && arr[i] != largest)
        {
            secondLargest = arr[i];
        }
    }

    return secondLargest;
}

    }
    printf("\n");
}


void find_and_display_duplicate(int arr[], int size)
{
    for (int i = 0; i < size; i++)
    {
        for (int j = i + 1; j < size; j++)
        {
            if (arr[i] == arr[j])
            {
                printf("Duplicate number %d found at location %d and %d\n", arr[i], i, j);

            }
        }
    }
}

{

    //Data Input

    //Processing

    while (!end_Reached)

        scanf("%c", &ch);

        if (ch == '*')

        else

            text[count++] = ch;

    }

    text[count] = '\0';

    //Output

    printf("The count of the characters is : %d\n", count);

    return 0;

/* A program to enter a text. Then enter a pattern and count the      */
/* number of times the pattern is repeated in the text. This program  */
/* will read in a string not a character at a time.                   */
/*								      */
/**********************************************************************/

#include <stdio.h>

int main()
{
    //Declaration

    char text[1000];
    char pattern[100];
    int count = 0;
    int i = 0;
    char ch;

    //Data/Input

    printf("Enter string: ");

    //Processing

    while ((ch = getchar()) != '\n' && i < 999)
    {
        text[i++] = ch;
    }
    text[i] = '\0';

    printf("Enter the pattern: ");

    i = 0;

    while ((ch = getchar()) != '\n' && i < 99)
    {
        pattern[i++] = ch;
    }
    pattern[i] = '\0';

    int j = 0;

    while (text[j] != '\0')
    {

        int match = 1;

        int k = 0;

        while (pattern[k] != '\0')
        {
            if (text[j + k] != pattern[k])
            {
                match = 0;
            }

            k++;
        }

        if (match)
        {
            count++;
            j += k;
        }
        else
        {
            j++;
        }
    }

    // Output

    if (count > 0)
    {
        printf("\nPattern found %d times\n", count);
    }
    else
    {
        printf("\nPattern not found\n");
    }

    return 0;
}
