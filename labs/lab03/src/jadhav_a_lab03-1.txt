{

}

float trigArea(float length, float height)
{
    return 0.5 * length * height;
}

void disp_Data(float length, float height, float area)
{
    printf("Triangle Length = %0.2f\n", length);
    printf("Triangle Height = %0.2f\n", height);
    printf("Triangle Area = %0.2f\n", area);
}

int main()
{
    float length, height, area;

    getData(&length, &height);
    area = trigArea(length, height);
    disp_Data(length, height, area);

    return 0;

#include<stdio.h>

//function prototype

double get_sales()
 {
    double sales;
    printf("Enter the monthly sales:");
    scanf("%lf", &sales);
    return sales;
}
double get_advanced_pay()
 {
    double advanced_pay;
    printf("Enter the amount of advanced pay, or enter 0 if no advanced pay was given.\n");
    printf("Advanced pay:");
    scanf("%lf", &advanced_pay);
    return (advanced_pay);
}
double determine_comm_rate(double sales)
{
    if (sales < 10000.00)
        {
            return 0.10;
        }
    else if (sales >= 10000.00 && sales <= 14999.99)
        {
            return 0.12;
        }
    else if (sales >= 15000.00 && sales <= 17999.99)
        {
            return 0.14;
        }
    else if (sales >= 18000.00 && sales <= 21999.99)
        {
            return 0.16;
        }
    else
        {
            return 0.18;
        }
}
int main()
{

 //Declaration

 double sales, advanced_pay, comm_rate, pay;

 //Call functions

 sales = get_sales();
 advanced_pay = get_advanced_pay();
 comm_rate = determine_comm_rate(sales);

 //Calculations

 pay = sales * comm_rate - advanced_pay;

 //Output

 printf("The pay is $%.2f", pay);
 if (pay < 0)
 {
    printf("\nThe salesperson must reimburse the company.");
 }

 return 0;

}
