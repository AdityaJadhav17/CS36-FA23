/**********************************************************************/   

#include<stdio.h>

int main()
{
    //Declaration

    int   rad_ius;
    float dia_meter;
    float area;
    float circum_ference;
    const float PI = 3.14159;

    //Data/Input

    printf("The radius of the circle is ");
    scanf("%d", &rad_ius);

    //Processing/Calculations

    dia_meter      = rad_ius * 2;
    area           = PI * rad_ius * rad_ius;
    circum_ference = 2 * PI * rad_ius;

    //Output

    printf("The diameter of the circle is %0.5f\n", dia_meter);
    printf("The area of the circle is %0.5f\n", area);
    printf("The circumference of the circle is %0.5f\n", circum_ference);

    return 0;

}
